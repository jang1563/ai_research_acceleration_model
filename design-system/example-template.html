# Calculate effective pipeline acceleration
def calculate_acceleration(stages):
    """
    The overall acceleration is limited by the bottleneck stage.
    This demonstrates the 'pipeline discount' effect.
    """
    traditional_total = sum(s['traditional'] for s in stages)
    ai_total = sum(s['ai_assisted'] for s in stages)

    return traditional_total / ai_total

# Example: Research pipeline
stages = [
    {'name': 'Literature Review', 'traditional': 21, 'ai_assisted': 2},
    {'name': 'Data Analysis', 'traditional': 180, 'ai_assisted': 14},
    {'name': 'Lab Work', 'traditional': 180, 'ai_assisted': 165},
    {'name': 'Clinical Trials', 'traditional': 1095, 'ai_assisted': 1095},
]

acceleration = calculate_acceleration(stages)
print(f"Effective acceleration: {acceleration:.1f}x")  # Output: 2.8x
